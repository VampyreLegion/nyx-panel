> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a FastAPI web app at `services.nyxstudios.net` (port 8085) that shows live status of all services across Nyx, Astraea, and Selene, with start/stop/restart/reboot controls.

**Architecture:** FastAPI on Nyx (port 8085) serves a single-page dashboard. The backend queries local systemd/docker via subprocess for Nyx services, and SSHes via paramiko for Astraea and Selene. The frontend polls `/api/status` every 5 seconds and diffs the result to update only changed service cards.

**Tech Stack:** Python 3, FastAPI, uvicorn, paramiko, Jinja2, vanilla JS, same CSS variables as nyx-step.

from __future__ import annotations
import subprocess
import logging

logger = logging.getLogger(__name__)

# ── Local (Nyx) ────────────────────────────────────────────────────────────────

def get_local_systemd_state(service: str) -> str:
    """Returns 'running', 'stopped', 'error', or 'unknown'."""
    result = subprocess.run(
        ["systemctl", "is-active", service],
        capture_output=True, text=True, timeout=5
    )
    raw = result.stdout.strip()
    if raw == "active":
        return "running"
    if raw in ("inactive", "deactivating"):
        return "stopped"
    if raw in ("failed", "activating"):
        return "error"
    return "unknown"


def get_local_docker_state(container: str) -> str:
    """Returns 'running', 'restarting', 'stopped', 'error', or 'unknown'."""
    result = subprocess.run(
        ["docker", "inspect", "--format", "{{.State.Status}}", container],
        capture_output=True, text=True, timeout=5
    )
    raw = result.stdout.strip()
    if raw == "running":
        return "running"
    if raw == "restarting":
        return "restarting"
    if raw in ("exited", "created", "dead"):
        return "stopped"
    if raw == "paused":
        return "stopped"
    return "unknown"
```

import paramiko
import contextlib


@contextlib.contextmanager
def _ssh_connect(host: str, user: str, password: str):
    client = paramiko.SSHClient()
    client.set_missing_host_key_policy(paramiko.AutoAddPolicy())
    try:
        client.connect(
            host, username=user, password=password,
            look_for_keys=False, allow_agent=False, timeout=8
        )
        yield client
    except Exception:
        yield None
    finally:
        client.close()


def run_ssh_command(host: str, user: str, password: str, cmd: str) -> str:
    """Run a command over SSH. Returns stdout string or empty string on failure."""
    try:
        with _ssh_connect(host, user, password) as ssh:
            if ssh is None:
                return ""
            _, stdout, _ = ssh.exec_command(cmd, timeout=10)
            return stdout.read().decode().strip()
    except Exception as e:
        logger.warning("SSH command failed %s@%s: %s", user, host, e)
        return ""


def get_remote_systemd_state(host: str, user: str, password: str, service: str) -> str:
    """Returns 'running', 'stopped', 'error', 'unreachable', or 'unknown'."""
    try:
        with _ssh_connect(host, user, password) as ssh:
            if ssh is None:
                return "unreachable"
            _, stdout, _ = ssh.exec_command(f"systemctl is-active {service}", timeout=10)
            raw = stdout.read().decode().strip()
    except Exception:
        return "unreachable"

    if raw == "active":
        return "running"
    if raw in ("inactive", "deactivating"):
        return "stopped"
    if raw in ("failed", "activating"):
        return "error"
    return "unknown"
```

def run_local_action(service: str, svc_type: str, action: str, sudo_password: str) -> tuple[bool, str]:
    """Run start/stop/restart on local machine. Returns (ok, error_message)."""
    if svc_type == "systemd":
        cmd = ["sudo", "-S", "systemctl", action, service]
        result = subprocess.run(cmd, input=sudo_password + "\n", capture_output=True, text=True, timeout=15)
    elif svc_type == "docker":
        cmd = ["docker", action, service]
        result = subprocess.run(cmd, capture_output=True, text=True, timeout=30)
    else:
        return False, f"Unknown service type: {svc_type}"

    if result.returncode != 0:
        return False, (result.stderr or "").strip()
    return True, ""


def run_remote_action(host: str, user: str, password: str, sudo_password: str,
                      service: str, svc_type: str, action: str) -> tuple[bool, str]:
    """Run start/stop/restart on remote machine via SSH."""
    if svc_type == "systemd":
        cmd = f"echo '{sudo_password}' | sudo -S systemctl {action} {service}"
    elif svc_type == "docker":
        cmd = f"docker {action} {service}"
    else:
        return False, f"Unknown service type: {svc_type}"

    try:
        result = run_ssh_command(host, user, password, cmd)
        return True, ""
    except Exception as e:
        return False, str(e)


def run_reboot(is_local: bool, host: str | None, user: str | None,
               password: str | None, sudo_password: str) -> tuple[bool, str]:
    """Reboot a machine. Returns (ok, error)."""
    if is_local:
        result = subprocess.run(
            ["sudo", "-S", "reboot"],
            input=sudo_password + "\n", capture_output=True, text=True, timeout=10
        )
        return result.returncode == 0, (result.stderr or "").strip()
    else:
        run_ssh_command(host, user, password, f"echo '{sudo_password}' | sudo -S reboot")
        return True, ""
```

# ── Status helpers ─────────────────────────────────────────────────────────────

def _get_machine_status(machine_id: str, machine: dict) -> dict:
    """Collect status for one machine synchronously (runs in thread pool)."""
    services = []
    reachable = True
    uptime = None

    if machine["is_local"]:
        # Get uptime locally
        import subprocess
        try:
            r = subprocess.run(["uptime", "-p"], capture_output=True, text=True, timeout=3)
            uptime = r.stdout.strip().replace("up ", "")
        except Exception:
            uptime = None

        for svc in machine["services"]:
            if svc["type"] == "systemd":
                state = get_local_systemd_state(svc["name"])
            elif svc["type"] == "docker":
                state = get_local_docker_state(svc["name"])
            else:
                state = "unknown"
            services.append({**svc, "state": state})
    else:
        host = machine["ip"]
        user = machine["ssh_user"]
        password = machine["ssh_password"]

        # Uptime via SSH (also tests reachability)
        uptime_raw = run_ssh_command(host, user, password, "uptime -p")
        if not uptime_raw:
            reachable = False
            uptime = None
        else:
            reachable = True
            uptime = uptime_raw.replace("up ", "")

        for svc in machine["services"]:
            if not reachable:
                state = "unreachable"
            elif svc["type"] == "systemd":
                state = get_remote_systemd_state(host, user, password, svc["name"])
            else:
                state = "unknown"
            services.append({**svc, "state": state})

    return {"reachable": reachable, "uptime": uptime, "services": services}


def fetch_all_status() -> dict:
    """Fetch status from all machines in parallel."""
    import concurrent.futures
    results = {}
    futures = {
        _pool.submit(_get_machine_status, mid, mcfg): mid
        for mid, mcfg in config.MACHINES.items()
    }
    for future in concurrent.futures.as_completed(futures):
        mid = futures[future]
        try:
            results[mid] = future.result(timeout=15)
        except Exception as e:
            logger.error("Status fetch failed for %s: %s", mid, e)
            results[mid] = {"reachable": False, "uptime": None, "services": []}
    return results


def execute_action(machine: str, service: str, svc_type: str, action: str) -> tuple[bool, str]:
    mcfg = config.MACHINES[machine]
    if mcfg["is_local"]:
        return run_local_action(service, svc_type, action, mcfg["sudo_password"])
    else:
        return run_remote_action(
            mcfg["ip"], mcfg["ssh_user"], mcfg["ssh_password"],
            mcfg["sudo_password"], service, svc_type, action
        )


def execute_reboot(machine: str) -> tuple[bool, str]:
    mcfg = config.MACHINES[machine]
    return run_reboot(
        is_local=mcfg["is_local"],
        host=mcfg["ip"],
        user=mcfg["ssh_user"],
        password=mcfg["ssh_password"],
        sudo_password=mcfg["sudo_password"],
    )

@router.get("/api/status")

async function poll() {
  try {
    const resp = await fetch("/api/status");
    if (!resp.ok) throw new Error(resp.status);
    const status = await resp.json();
    _lastStatus = status;
    renderAll(status);
    const el = document.getElementById("refresh-status");

async function doAction(machine, service, type, action) {
  try {
    const resp = await fetch("/api/action", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ machine, service, type, action }),
    });
    const data = await resp.json();
    if (!data.ok) {
      showToast("Error: " + (data.error || "unknown"), "error");
    } else {
      showToast(`${action} sent to ${service}`, "success");
      setTimeout(poll, 1500);
    }
  } catch (e) {
    showToast("Request failed: " + e.message, "error");
  }
}

      body: JSON.stringify({ machine }),
    });
    const data = await resp.json();
    if (!data.ok) {
      showToast("Reboot failed: " + (data.error || "unknown"), "error");
    } else {
      showToast(`Rebooting ${machine}…`, "success");
    }
  } catch (e) {
    showToast("Request failed: " + e.message, "error");
  }
}

    "ps aux --sort=-%mem "
    "--format='%U %p %.1%C %.0rss %a' "
    "| head -21"
)
_TOP_CPU_CMD = (
    "ps aux --sort=-%cpu "
    "--format='%U %p %.1%C %.0rss %a' "
    "| head -21"
)
_PORTS_CMD = "ss -tlnp | grep LISTEN"
_CRON_CMD = (
    "crontab -l 2>/dev/null; "
    "for f in /etc/cron.d/* /etc/cron.daily/* /etc/cron.hourly/* /etc/cron.weekly/* /etc/cron.monthly/*; "
    "do echo \"=== $f ===\"; cat \"$f\" 2>/dev/null; done"
)
_TIMERS_CMD = "systemctl list-timers --no-pager 2>/dev/null | head -30"
_UPTIME_CMD = "uptime"


def _collect_local() -> dict:
    import subprocess
    def run(cmd):
        try:
            r = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=10)
            return r.stdout.strip()
        except Exception as e:
            return f"Error: {e}"

    return {
        "memory":  run(_MEMORY_CMD),
        "top_mem": run(_TOP_MEM_CMD),
        "top_cpu": run(_TOP_CPU_CMD),
        "ports":   run(_PORTS_CMD),
        "cron":    run(_CRON_CMD),
        "timers":  run(_TIMERS_CMD),
        "uptime":  run(_UPTIME_CMD),
    }


def _collect_remote(host: str, user: str, password: str) -> dict:
    def run(cmd):
        return run_ssh_command(host, user, password, cmd) or "Unavailable"

    return {
        "memory":  run(_MEMORY_CMD),
        "top_mem": run(_TOP_MEM_CMD),
        "top_cpu": run(_TOP_CPU_CMD),
        "ports":   run(_PORTS_CMD),
        "cron":    run(_CRON_CMD),
        "timers":  run(_TIMERS_CMD),
        "uptime":  run(_UPTIME_CMD),
    }


def _collect_for_machine(machine_id: str) -> dict:
    mcfg = config.MACHINES[machine_id]
    if mcfg["is_local"]:
        return _collect_local()
    return _collect_remote(mcfg["ip"], mcfg["ssh_user"], mcfg["ssh_password"])


@router.get("/api/sysinfo/{machine}")
async def api_sysinfo(machine: str):
    if machine not in config.MACHINES:
        return JSONResponse({"error": "Unknown machine"}, status_code=400)
    loop = asyncio.get_event_loop()
    data = await loop.run_in_executor(_pool, lambda: _collect_for_machine(machine))
    return data
```

```javascript
// ── Deep Dive ─────────────────────────────────────────────────────────────────

let _diveData = null;
let _diveSection = "memory";

async function openDive(machine, label) {
  const modal = document.getElementById("dive-modal");
  const title = document.getElementById("dive-title");
  const pre   = document.getElementById("dive-pre");
  title.textContent = `🔍 ${label} — System Deep Dive`;
  pre.textContent = "Fetching data…";
  modal.classList.remove("hidden");
  _diveData = null;
  _diveSection = "memory";
  document.querySelectorAll(".dive-tab").forEach(t => {
    t.classList.toggle("active", t.dataset.section === "memory");
  });

  try {
    const resp = await fetch(`/api/sysinfo/${machine}`);
    if (!resp.ok) throw new Error(resp.status);
    _diveData = await resp.json();
    pre.textContent = _diveData[_diveSection] || "(empty)";
  } catch (e) {
    pre.textContent = "Error: " + e.message;
  }
}

function closeDive() {
  document.getElementById("dive-modal").classList.add("hidden");
  _diveData = null;
}

function showDiveSection(btn, section) {
  _diveSection = section;
  document.querySelectorAll(".dive-tab").forEach(t => t.classList.remove("active"));
  btn.classList.add("active");
  const pre = document.getElementById("dive-pre");
  if (_diveData) {
    pre.textContent = _diveData[section] || "(empty)";
  }
}

// Close on backdrop click
document.addEventListener("DOMContentLoaded", () => {
  document.getElementById("dive-modal").addEventListener("click", (e) => {
    if (e.target.id === "dive-modal") closeDive();
  });
});
```
